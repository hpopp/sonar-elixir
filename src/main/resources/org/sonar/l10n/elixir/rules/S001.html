Compliant Solution

def process(data) do
  with :ok <- validate(data),
       result <- classify_and_handle(data) do
    result
  end
end

defp classify_and_handle(data) do
  case classify(data) do
    :type_a -> handle_type_a(data)
    :type_b -> handle_type_b(data)
  end
end
